48  /**
49   * Unit tests for {@link RequestLimitFilter}
50   *
51   * @author rlichti {@literal <rlichti@kaiserpfalz-edv.de>}
52   * @since 3.0.0 2023-01-17
53   */
54  @SpringBootTest(
55      webEnvironment = WebEnvironment.DEFINED_PORT,
56      classes = {
57          EanSearchClient.class,
58          EanSearchClientConfig.class,
59          RequestLimitFilter.class,
60          ResponseErrorMapper.class
61      }
62  )
63  @EnableAutoConfiguration(exclude = {DataSourceAutoConfiguration.class, DataSourceTransactionManagerAutoConfiguration.class, HibernateJpaAutoConfiguration.class})
64  @EnableFeignClients(clients = {EanSearchClient.class})
65  @AutoConfigureWireMock(port = 8089)
66  @Slf4j
67  public class EanSearchClientTest  extends AbstractTestBase {
68  
69      @Autowired
70      private EanSearchClient sut;
71  
72      @Autowired
73      private RequestLimitFilter filter;
74  
75      @PostConstruct
76      public void init() {
77          super.setLog(log);
78          super.setTestSuite("library-lookup");
79      }
80  
81      @Test
82      void shouldIncrementTheCounterInFilterWhenRequestIsDone() {
83          try {
84              final Set<EanData> result = this.sut.barcodeLookupEAN("978-5-01234-678-9");
85              log.info("Result. result={}", result);
86  
87              assertEquals(1, result.size());
88              assertEquals(15, this.filter.getRemaining());
89          } catch (final EanSearchException e) {
90              Assertions.fail(e);
91          }
92      }
93  
94      @Test
95      void shouldThrowExceptionWhenNoCreditIsLeft() {
96          final EanSearchTooManyRequestsException thrown = Assertions.assertThrows(EanSearchTooManyRequestsException.class, () -> {
97              this.sut.barcodeLookupEAN("978-5-01234-678-1");
98          });
99  
100         assertEquals("Too many requests (eg. rate limit exceeded).", thrown.getMessage());
101     }
102 
103     @Test
104     void shouldThrowExceptionWhenCreditsRanOutDuringRequest() {
105         this.sut.barcodeLookupEAN("978-5-01234-678-0"); // used the last credit.
106 
107         final EanSearchTooManyRequestsException thrown = Assertions.assertThrows(EanSearchTooManyRequestsException.class, () -> {
108             this.sut.barcodeLookupEAN("123412312"); // irrelevant query - should be filtered ...
109         });
110 
111         assertEquals("Too many requests (eg. rate limit exceeded).", thrown.getMessage());
112     }
113 
114     @Test
115     void shouldReturnTheDefaultEntryWhenCalledWithISBN() {
116         final Set<EanData> result = this.sut.barcodeLookupISBN("5-01234-678-9");
117 
118         assertEquals(1, result.size());
119     }
120 
121     @BeforeEach
122     void resetFilter() {
123         this.filter.reset();
124     }
125 }
