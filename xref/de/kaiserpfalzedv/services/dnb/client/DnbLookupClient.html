18  package de.kaiserpfalzedv.services.dnb.client;
19  
20  import java.util.List;
21  
22  import org.springframework.cloud.openfeign.FeignClient;
23  import org.springframework.web.bind.annotation.RequestMapping;
24  import org.springframework.web.bind.annotation.RequestMethod;
25  import org.springframework.web.bind.annotation.RequestParam;
26  
27  import de.kaiserpfalzedv.services.dnb.model.Book;
28  
29  /**
30   * <p>DnbLookupClient -- The client for accessing the webservice.</p>
31   *
32   * @author rlichti {@literal <rlichti@kaiserpfalz-edv.de>}
33   * @since 3.0.0  2023-01-17
34   */
35  @FeignClient(name = "dnb-lookup", configuration = DnbLookupClientConfig.class)
36  public interface DnbLookupClient {
37      /**
38       * This is the generic query to the DNB index.
39       *
40       * <p>Will result in something like:</p>
41       * <pre>https://services.dnb.de/sru/dnb?version=1.1&operation=searchRetrieve&query=WOE%3D9783958672567&recordSchema=MARC21-xml</pre>
42       *
43       * @param query The query to be sent. It has to contain the index followed by a '='. The nicest index is 'WOE'.
44       * @return A set of Books.
45       */
46      @RequestMapping(method = RequestMethod.GET, value = "/sru/dnb", produces = "text/xml", consumes = "text/xml")
47      List<Book> lookup(@RequestParam("query") String query);
