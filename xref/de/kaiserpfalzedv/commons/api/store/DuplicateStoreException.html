18  package de.kaiserpfalzedv.commons.api.store;
19  
20  import de.kaiserpfalzedv.commons.api.BaseSystemException;
21  import de.kaiserpfalzedv.commons.api.resources.Pointer;
22  
23  /**
24   * DuplicateStoreException -- There is already an object with this metadata.
25   *
26   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
27   * @since 2.0.0  2021-05-24
28   */
29  public class DuplicateStoreException extends BaseSystemException {
30      private final Pointer stored;
31      private final Pointer duplicate;
32  
33      /**
34       * @param stored    the already stored resource metadata.
35       * @param duplicate the new resource metadata.
36       * @since 2.0.0  2021-05-24
37       */
38      public DuplicateStoreException(final Pointer stored, final Pointer duplicate) {
39          super(String.format("Duplicate element found. resource='%s', nameSpace='%s', name='%s'",
40                  stored.getKind(), stored.getNameSpace(), stored.getName()
41          ));
42  
43          this.stored = stored;
44          this.duplicate = duplicate;
45      }
46  
47      /**
48       * @return the generation stored in the data store.
49       */
50      public Pointer getStoredGeneration() {
51          return this.stored;
52      }
53  
54      /**
55       * @return the generation that should be saved.
56       */
57      public Pointer getSaveGeneration() {
58          return this.duplicate;
59      }
60  }
