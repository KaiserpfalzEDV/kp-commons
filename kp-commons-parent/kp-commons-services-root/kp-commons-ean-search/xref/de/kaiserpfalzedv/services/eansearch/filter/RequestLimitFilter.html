18  package de.kaiserpfalzedv.services.eansearch.filter;
19  
20  import java.time.OffsetDateTime;
21  import java.time.ZoneOffset;
22  import java.time.temporal.ChronoUnit;
23  
24  import org.springframework.beans.factory.annotation.Autowired;
25  
26  import de.kaiserpfalzedv.services.eansearch.mapper.EanSearchException;
27  import de.kaiserpfalzedv.services.eansearch.mapper.EanSearchTooManyRequestsException;
28  import feign.InvocationContext;
29  import feign.RequestInterceptor;
30  import feign.RequestTemplate;
31  import feign.ResponseInterceptor;
32  import io.micrometer.core.instrument.Counter;
33  import io.micrometer.core.instrument.MeterRegistry;
34  import io.micrometer.core.instrument.Tags;
35  import jakarta.annotation.PostConstruct;
36  import jakarta.annotation.PreDestroy;
37  import jakarta.inject.Singleton;
38  import lombok.Getter;
39  import lombok.RequiredArgsConstructor;
40  import lombok.extern.slf4j.Slf4j;
41  
42  /**
43   * <p>RequestLimitFilter -- Reports the used credits and the remaining credits for the EAN search api.</p>
44   *
45   * @author klenkes74 {@literal <rlichti@kaiserpfalz-edv.de>}
46   * @since 1.0.0  2023-01-17
47   */
48  @Singleton
49  @RequiredArgsConstructor(onConstructor = @__(@Autowired))
50  @Slf4j
51  public class RequestLimitFilter implements RequestInterceptor, ResponseInterceptor {
52      private static final String API_REMAINING_REQUEST_HEADER = "X-Credits-Remaining";
53      private static final String API_REMAINING_METRICS_NAME = "ean-search.credits.remaining";
54      private static final String API_REQUESTS_HANDLED_SINCE_START = "ean-search.credits.used-since-start";
55  
56      private final MeterRegistry registry;
57      private Counter requestCounter;
58  
59      @Getter
60      private int remaining = -1;
61      private OffsetDateTime lastRequest = OffsetDateTime.now(ZoneOffset.UTC).minus(1, ChronoUnit.DAYS);
62  
63      @PostConstruct
64      public void registerMetric() {
65          this.requestCounter = this.registry.counter(API_REQUESTS_HANDLED_SINCE_START, Tags.empty());
66      }
67  
68      @PreDestroy
69      public void unregisterMetrics() {
70          this.requestCounter.close();
71      }
72  
73      /**
74       * Resets the counter and enables new requests, that would otherwise being blocked by this filter.
75       */
76      public synchronized void reset() {
77          this.remaining = -1;
78          log.info("EAN Search credit reporter reset. Requests should now result in requests to the API.");
79      }
80  
81  
82  
83      /**
84       * Reports the handled requests and remaining credits to the API.
85       *
86       * @param requestContext  request context.
87       * @param responseContext response context.
88       */
89  
90      @Override
91      public Object intercept(final InvocationContext context, final Chain chain) throws Exception {
92          final String remaining = context.response().headers().get(API_REMAINING_REQUEST_HEADER).stream().findFirst().orElse(null);
93  
94          if (remaining != null) {
95              synchronized (this) {
96                  this.remaining = Integer.valueOf(remaining, 10);
97                  this.registry.gauge(API_REMAINING_METRICS_NAME, Tags.empty(), this.remaining);
98              }
99          }
100         this.requestCounter.increment();
101 
102         log.debug("EAN-Search remaining requests. remaining={}, used={}", this.remaining, this.requestCounter.count());
103         return chain.next(context);
104     }
105 
106     /**
107      * This filter prevents the external call when there are no credits left for this API.
108      *
109      * @param requestContext request context.
110      * @throws EanSearchException When the credits for this API has been used and there are no
111      *                                               credits remaining.
112      */
113     @Override
114     public void apply(final RequestTemplate request) throws EanSearchException {
115         if (
116                 this.remaining == 0
117                 && this.lastRequestWasToday()
118         ) {
119             log.error("Can't search for EAN any more. There is no remaining credit left. lastRequest='{}',", this.lastRequest);
120 
121             throw new EanSearchTooManyRequestsException();
122         }
123 
124         synchronized (this) {
125             this.lastRequest = OffsetDateTime.now(ZoneOffset.UTC);
126         }
127     }
128 
129     private boolean lastRequestWasToday() {
130         return this.lastRequest.isAfter(
131                 OffsetDateTime.now(ZoneOffset.UTC).toLocalDate()
132                         .atStartOfDay(ZoneOffset.UTC)
133                         .toOffsetDateTime()
134         );
135     }
136 }
